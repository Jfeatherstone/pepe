 12@numba.jit(nopython=True)
 13def crossMask(size, ylim=[0,0], xlim=[0,0], channels=3):
 14    """
 15    Create a mask in the shape of a cross, in which the inside of the cross is
 16    the desired region.
 17
 18    A stripe (horizontal or vertical) can be made by only supplying a single set
 19    of limits, either for x (vertical) or y (horizontal).
 20
 21    Parameters
 22    ----------
 23
 24    size : [int, int] or [int, int, int]
 25        The size of the mask to be created (height, width[, channels]). Can include number of channels.
 26        If not included, default is 3; see `channels`.
 27
 28    ylim : [int, int]
 29        The y coordinates of the (beginning and end of the) desired horizontal region
 30
 31    xlim : [int, int]
 32        The x coordinates of the (beginning and end of the) desired vertical region
 33
 34    channels : int
 35        The number of channels to create for the mask, if not included in `size` parameter.
 36
 37    Returns
 38    -------
 39
 40    np.uint8[3] : Mask array (height, width, channels)
 41    """
 42
 43    # If channels are included in the size, use that
 44    if len(size) == 3:
 45        channels = size[2]
 46
 47    maskArr = np.zeros((size[0], size[1], channels), dtype=np.uint8)
 48
 49    for i in range(size[0]):
 50        for j in range(size[1]):
 51            if (i >= ylim[0] and i <= ylim[1]):
 52                maskArr[i,j] = np.ones(channels)
 53            elif (j >= xlim[0] and j <= xlim[1]):
 54                maskArr[i,j] = np.ones(channels)
 55
 56    return maskArr

 59@numba.jit(nopython=True)
 60def rectMask(size, corner=[0,0], dimensions=[0,0], channels=3):
 61    """
 62    Create a mask in the shape of a (non-rotated) rectangle, in which the inside
 63    of the rectangle is the desired region.
 64
 65    Parameters
 66    ----------
 67
 68    size : [int, int] or [int, int, int]
 69        The size of the mask to be created (height, width[, channels]). Can include
 70        number of channels. If not included, default is 3; see `channels`.
 71
 72    corner : [int, int]
 73        The [y, x] coordinates of the top-left corner of the desired region. 
 74
 75    dimensions : [int, int]
 76        The [height, width] of the desired region. If either is set to 0, the
 77        entire rest of the image will be included in that direction.
 78
 79    channels : int
 80        The number of channels to create for the mask, if not included in `size` parameter.
 81
 82    Returns
 83    -------
 84
 85    np.uint8[3] : Mask array (height, width, channels)
 86    """
 87
 88    # If channels are included in the size, use that
 89    # Though if the channels variable is explicitly set to None, that overrides this
 90    if len(size) == 3:
 91        channels = size[2]
 92
 93    xlim = [corner[1], corner[1] + dimensions[1]]
 94    ylim = [corner[0], corner[0] + dimensions[0]]
 95
 96    maskArr = np.zeros((size[0], size[1], channels), dtype=np.uint8)
 97    
 98    for i in range(size[0]):
 99        for j in range(size[1]):
100            if (i >= ylim[0] and i <= ylim[1] and j >= xlim[0] and j <= xlim[1]):
101                maskArr[i,j] = np.ones(channels)
102
103    return maskArr

105@numba.jit(nopython=True)
106def circularMask(size, center, radius, channels=3):
107    """
108    Create a mask in the shape of a circle, in which the inside of the circle is
109    the desired region.
110
111    Parameters
112    ----------
113
114    size : [int, int] or [int, int, int]
115        The size of the mask to be created (height, width[, channels]). Can include number of channels.
116        If not included, default is 3; see `channels`.
117
118    center : [int, int]
119        The [y, x] coordinates of the center of the desired region. 
120
121    radius : int
122        The radius of the desired region.
123
124    channels : int
125        The number of channels to create for the mask, if not included in `size` parameter.
126
127    Returns
128    -------
129
130    np.uint8[3] : Mask array (height, width, channels)
131    """
132
133    # If channels are included in the size, use that
134    if len(size) == 3:
135        channels = size[2]
136
137    # numba doesn't support ogrid, so we just do it manually
138    #Y, X = np.ogrid[:size[0], :size[1]]
139    Y = np.arange(size[0]).reshape((size[0], 1)) # Column vector
140    X = np.arange(size[1]).reshape((1, size[1])) # Row vector
141
142    dist_from_center = np.sqrt((X - center[1])**2 + (Y-center[0])**2)
143
144    maskArr = np.zeros((size[0], size[1], channels), np.uint8)
145    for i in range(channels):
146        maskArr[:,:,i] = dist_from_center <= radius
147
148    return maskArr

151@numba.jit(nopython=True)
152def ellipticalMask(size, majorAxisEndpoint1, majorAxisEndpoint2, minorAxisLength, channels=3):
153    """
154    Create a mask in the shape of a circle, in which the inside of the circle is
155    the desired region.
156
157    Parameters
158    ----------
159
160    size : [int, int] or [int, int, int]
161        The size of the mask to be created (height, width[, channels]). Can include number of channels.
162        If not included, default is 3; see `channels`.
163
164    majorAxisEndpoint1 : [int, int]
165        The first of two sets of [y, x] coordinates that define the major axis of the desired region. 
166        Order does not matter.
167
168    majorAxisEndpoint2 : [int, int]
169        The second of two sets of [y, x] coordinates that define the major axis of the desired region. 
170        Order does not matter.
171
172    minorAxisLength : int
173        The length of the minor axis, or the extent of the shorter dimension of the ellipse.
174
175    channels : int
176        The number of channels to create for the mask, if not included in `size` parameter.
177
178    Returns
179    -------
180
181    np.uint8[3] : Mask array (height, width, channels)
182    """
183
184    minorAxisLength = float(minorAxisLength)
185
186    # If channels are included in the size, use that
187    # Though if the channels variable is explicitly set to None, that overrides this
188    if len(size) == 3:
189        channels = size[2]
190
191    # Center of ellipse
192    center = np.array([(majorAxisEndpoint1[0] + majorAxisEndpoint2[0])/2,
193              (majorAxisEndpoint1[1] + majorAxisEndpoint2[1])/2], dtype=np.float64)
194
195    # Unit vector along major axis
196    eMajor = np.array([majorAxisEndpoint1[0] - majorAxisEndpoint2[0],
197                       majorAxisEndpoint1[1] - majorAxisEndpoint2[1]], dtype=np.float64)
198    eMajor /= np.sqrt(eMajor[0]**2 + eMajor[1]**2)
199
200    semiMajorAxis = np.sqrt((majorAxisEndpoint1[0] - majorAxisEndpoint2[0])**2 + (majorAxisEndpoint1[1] - majorAxisEndpoint2[1])**2)/2
201    semiMinorAxis = minorAxisLength/2
202
203    # Distance from center to either focus part of a triangle with axes
204    focalDistance = np.sqrt(semiMajorAxis**2 - semiMinorAxis**2)
205
206    # Foci of the ellipse
207    foci = np.zeros((2, 2))
208    foci[0,0] = center[0] + focalDistance * eMajor[0]
209    foci[0,1] = center[1] + focalDistance * eMajor[1]
210    foci[1,0] = center[0] - focalDistance * eMajor[0]
211    foci[1,1] = center[1] - focalDistance * eMajor[1]
212    #foci = [[center[0] + focalDistance * eMajor, center[1] + focalDistance * eMajor],
213    #        [center[0] - focalDistance * eMajor, center[1] - focalDistance * eMajor]]
214
215    # For any point outside of the ellipse, the sum of distance from both foci will
216    # be greater than the length of the major axis
217
218    # numba doesn't support ogrid, so we just do it manually
219    #Y, X = np.ogrid[:size[0], :size[1]]
220    Y = np.arange(size[0]).reshape((size[0], 1)) # Column vector
221    X = np.arange(size[1]).reshape((1, size[1])) # Row vector
222
223    fociDistanceSum = np.sqrt((X - foci[0][1])**2 + (Y - foci[0][0])**2) + np.sqrt((X - foci[1][1])**2 + (Y - foci[1][0])**2) 
224    
225    maskArr = np.zeros((size[0], size[1], channels), dtype=np.uint8)
226    for i in range(channels):
227        maskArr[:,:,i] = fociDistanceSum <= 2*semiMajorAxis
228
229    return maskArr

232@numba.jit(nopython=True)
233def mergeMasks(listOfMasks, signs=None, channels=3):
234    """
235    Merge several masks together into a single new mask, serving as the union of the
236    previous set.
237
238    Parameters
239    ----------
240
241    listOfMasks : iterable of np.uint8[2] or np.uint8[3]
242        A list of the masks to be merged together. If they have different numbers of channels, the
243        union will follow the `channels` parameter; otherwise, original shape will be preserved.
244
245    signs : iterable of int
246        Array of 1 or -1 that indicates which masks should be added to the final result, and which should
247        be subtracted. By default, all are added.
248
249    channels : int
250        The number of channels to create for the final mask; used only if not consistent in `listOfMasks` parameter.
251
252    Returns
253    -------
254
255    np.uint8[2] or np.uint8[3] : Mask array; can include channels in last dim, or not.
256    """
257
258    if signs is None:
259        signs = np.ones(len(listOfMasks), dtype=np.int16)
260
261    sumMask = np.zeros((*np.shape(listOfMasks[0])[:2], channels), dtype=np.int16)
262
263    for i in range(len(listOfMasks)):
264        for j in range(channels):
265            # Assume shape of [H, W, C]
266            sumMask[:,:,j] += listOfMasks[i][:,:,0] * signs[i]
267                
268
269    #sumMask += np.abs(np.min(sumMask))
270    #sumMask = np.ceil(sumMask / np.max(sumMask))
271    return (sumMask > 0).astype(np.uint8)

274def identifyFeatureBoundary(image,
275                            approxFeatureLocation,
276                            featureSize,
277                            meanThresholdFactor=2,
278                            minIslandSize=5,
279                            pointDownsampleFactor=5,
280                            concaveHullParameter=0,
281                            debug=False):
282    """
283    Detect the boundary of a feature near the specified location.
284
285    Feature detection is done using the same algorithm as in
286    `pepe.preprocess.maskFeature()`. Feature should be
287    brighter than the background.
288
289    Parameters
290    ----------
291    image : numpy.ndarray[H,W[,C]]
292        Input image to identify feature in. If multi-channel,
293        the channels will be averaged (gray-scaled).
294
295    approxFeatureLocation : (float, float)
296        Approximate coordinates (y,x) of the location of the feature.
297
298    meanThresholdFactor : float
299        Parameter for determining which pixels are a part of the feature.
300        Lower value means the detected feature will be larger.
301
302    minIslandSize : int
303        Parameter for culling pixels initially detected to be a part of
304        the feature, but likely are not actually. eg. If a value of 5,
305        is chosen, only pixels surrounded by 24 other pixels detected to
306        be a part of the feature will be kept.
307
308        If not an odd number, the value will be rounded up to the next odd
309        number.
310
311    pointDownsampleFactor : int >= 1
312        Parameter for generating the hull around the points detected to be
313        a part of the feature. Higher value means more approximate shape,
314        but faster calculations. Value of `1` will give the most accurate
315        result.
316
317    concaveHullParamter : float
318        Alphashape parameter for generating the hull. A value of `0`
319        means the hull will be convex, a value `>= 0` will mean the
320        hull is increasingly concave.
321
322        For more information, see [alphashape documentation](https://alphashape.readthedocs.io/en/latest/readme.html)
323        related to the 'alpha parameter'.
324
325    debug : bool
326        Whether to plot debug information about the feature
327        detection once completed or not.
328
329
330    Returns
331    -------
332    boundaryPoints : numpy.ndarray[N,2]
333        List of N points that comprise the boundary ordered
334        such that the boundary can be drawn by drawing all pairs
335        of points (i,i+1) (and (-1,0) for the final line).
336    """
337    # Grayscale if necessary
338    if len(np.shape(image)) == 3:
339        grayImage = np.mean(image, axis=-1)
340    else:
341        grayImage = image
342
343    # Crop
344    croppedImage = grayImage[approxFeatureLocation[0]-int(featureSize):approxFeatureLocation[0]+int(featureSize),
345                                approxFeatureLocation[1]-int(featureSize):approxFeatureLocation[1]+int(featureSize)]
346
347    # Identify regions that are above the mean
348    thresholdedImage = np.float64(croppedImage > meanThresholdFactor*np.mean(croppedImage))
349
350    # Blur and then threshold again, meaning that anything who isn't surrounded by
351    # `minIslandSize/2` neighbors in each direction gets removed
352    kernel = minIslandSize + int(not minIslandSize%2) # Make sure it is odd
353    thresholdedImage = np.float64(cv2.blur(thresholdedImage, (kernel,kernel)) == 1)
354
355    # Generate a convex hull
356    alphashape = ap.alphashape(np.array(np.where(thresholdedImage)).T[::pointDownsampleFactor], concaveHullParameter)
357    
358    boundaryPoints = np.array(alphashape.boundary.coords)[:,:2:][:,::-1]
359
360    if debug:
361        fig, ax = plt.subplots(1, 2, figsize=(8,4))
362
363        ax[0].imshow(thresholdedImage)
364        ax[1].imshow(grayImage)
365        for i in range(len(boundaryPoints-1)):
366            ax[1].plot(boundaryPoints[i:i+2,0], boundaryPoints[i:i+2,1], c='red', alpha=1)
367
368        ax[0].set_title('Thresholded Image')
369        ax[1].set_title('Identified Feature Boundary')
370        plt.show()
371
372    # Convert local coordinates back to absolute image coordinates
373    return approxFeatureLocation + boundaryPoints - featureSize

375def maskFeature(image,
376                approxFeatureLocation,
377                featureSize,
378                meanThresholdFactor=1,
379                minIslandSize=5,
380                pointDownsampleFactor=5,
381                concaveHullParameter=0,
382                extraPadding=0,
383                allowRecurse=True,
384                debug=False):
385    """
386    Generate a mask to remove a detected feature near the specified location.
387
388    Feature detection is done using the same algorithm as in
389    `pepe.preprocess.detectFeatureBoundary()`. Feature should be
390    brighter than the background.
391
392    Parameters
393    ----------
394    image : numpy.ndarray[H,W[,C]]
395        Input image to identify feature in. If multi-channel,
396        the channels will be averaged (gray-scaled).
397
398    approxFeatureLocation : (float, float)
399        Approximate coordinates (y,x) of the location of the feature.
400
401    meanThresholdFactor : float
402        Parameter for determining which pixels are a part of the feature.
403        Lower value means the detected feature will be larger.
404
405    minIslandSize : int
406        Parameter for culling pixels initially detected to be a part of
407        the feature, but likely are not actually. eg. If a value of 5,
408        is chosen, only pixels surrounded by 24 other pixels detected to
409        be a part of the feature will be kept.
410
411        If not an odd number, the value will be rounded up to the next odd
412        number.
413
414    pointDownsampleFactor : int >= 1
415        Parameter for generating the hull around the points detected to be
416        a part of the feature. Higher value means more approximate shape,
417        but faster calculations. Value of `1` will give the most accurate
418        result.
419
420    concaveHullParamter : float
421        Alphashape parameter for generating the hull. A value of `0`
422        means the hull will be convex, a value `>= 0` will mean the
423        hull is increasingly concave.
424
425        For more information, see [alphashape documentation](https://alphashape.readthedocs.io/en/latest/readme.html)
426        related to the 'alpha parameter'.
427
428    extraPadding : int
429        Extra padding at the end of the feature identification
430        added to the shape.
431
432    allowRecurse : bool
433        If the  detected feature is larger than the size,
434        and this value is `True`, the function will recursively
435        call itself (at most once) with a larger feature size to try
436        and detect the entire feature.
437        
438    debug : bool
439        Whether to plot debug information about the feature
440        detection once completed or not.
441
442
443    Returns
444    -------
445    imageMask : numpy.ndarray[H,W]
446        Image mask with values of `1` everywhere
447        except for points that are detected to be a
448        part of the feature, which will have values
449        of `0`.
450
451    """
452    # Grayscale if necessary
453    if len(np.shape(image)) == 3:
454        grayImage = np.mean(image, axis=-1)
455    else:
456        grayImage = image
457
458    # Crop
459    croppedImage = grayImage[max(approxFeatureLocation[0]-int(featureSize), 0):min(approxFeatureLocation[0]+int(featureSize), grayImage.shape[0]),
460                                max(approxFeatureLocation[1]-int(featureSize), 0):min(approxFeatureLocation[1]+int(featureSize), grayImage.shape[1])]
461
462    # Identify regions that are above the mean
463    thresholdedImage = np.float64(croppedImage > meanThresholdFactor*np.mean(croppedImage))
464
465    # Blur and then threshold again, meaning that anything who isn't surrounded by
466    # `minIslandSize/2` neighbors in each direction gets removed
467    kernel = minIslandSize + int(not minIslandSize%2) # Make sure it is odd
468    thresholdedImage = np.float64(cv2.blur(thresholdedImage, (kernel,kernel)) == 1)
469
470    # Generate a convex hull
471    alphashape = ap.alphashape(np.array(np.where(thresholdedImage)).T[::pointDownsampleFactor], concaveHullParameter)
472
473    if alphashape is not None:
474        boundaryPoints = np.array(alphashape.boundary.coords)[:,:2:][:,::-1]
475    else:
476        boundaryPoints = None
477
478    # If any of the boundary points are at the edge of the image, that means the 
479    # feature extends beyond the cropped region, and we should try again with a slightly
480    # larger region
481    edgePadding = 5
482    if (True in (boundaryPoints.flatten() < edgePadding) or True in (boundaryPoints.flatten() + edgePadding > 2*featureSize)) and allowRecurse:
483        # We could try and recenter the crop window, but that doesn't actually work all that well...
484        # Maybe since averaging the boundary points doesn't actually calculate the center of mass
485        #newFeatureCenter = approxFeatureLocation + np.int64(np.mean(boundaryPoints, axis=0)) - featureSize
486        newFeatureCenter = approxFeatureLocation
487        return maskFeature(image, newFeatureCenter, featureSize*1.5,
488                           meanThresholdFactor, minIslandSize, pointDownsampleFactor,
489                           concaveHullParameter, False, debug)
490        
491    maskImage = np.zeros_like(croppedImage, dtype=np.uint8)
492
493    for i in range(len(boundaryPoints)-1):
494        cv2.line(maskImage, np.int64(boundaryPoints[i]), np.int64(boundaryPoints[i+1]), 1, minIslandSize)        
495    cv2.line(maskImage, np.int64(boundaryPoints[-1]), np.int64(boundaryPoints[0]), 1, minIslandSize)
496                    
497    outsidePoint = (0,0)
498    cv2.floodFill(maskImage, None, outsidePoint, 2)
499
500    # Add extra padding
501    if extraPadding > 0:
502        oddPadding = extraPadding + int(not extraPadding%2) # Make sure it is odd
503        maskImage = cv2.blur(maskImage, (oddPadding, oddPadding))
504
505    maskImage = maskImage == 2
506
507    fullImageMask = np.ones(image.shape[:2], dtype=np.uint8)
508    fullImageMask[max(approxFeatureLocation[0]-int(featureSize), 0):min(approxFeatureLocation[0]+int(featureSize), grayImage.shape[0]),
509                                max(approxFeatureLocation[1]-int(featureSize), 0):min(approxFeatureLocation[1]+int(featureSize), grayImage.shape[1])] = maskImage
510
511    if debug:
512        fig, ax = plt.subplots(1, 3, figsize=(12,4))
513
514        ax[0].imshow(thresholdedImage)
515        ax[1].imshow(croppedImage)
516        for i in range(len(boundaryPoints)-1):
517            ax[1].plot(boundaryPoints[i:i+2,0], boundaryPoints[i:i+2,1], c='red', alpha=1)
518
519        ax[2].imshow(fullImageMask)
520        
521        ax[0].set_title('Thresholded Image')
522        ax[1].set_title('Identified Feature Boundary')
523        ax[2].set_title('Full Image Mask')
524        plt.show()
525
526    return fullImageMask

529def polygonMask(boundaryPoints, maskSize, debug=False):
530    """
531    Create a mask of a polygon defined by the points that
532    comprise its boundary.
533
534    Parameters
535    ----------
536    boundaryPoints : iterable of points
537        Iterable of points (y,x) that define the
538        boundary of the polygon. Should be ordered
539        such that drawing each pair of points (i,i+1)
540        creates the desired shape.
541
542    maskSize : [H,W]
543        The size of the mask array to create.
544
545    debug : bool
546        Whether or not to plot a visualization of
547        the mask once it is completed.
548
549    Returns
550    -------
551    maskImage : numpy.ndarray[H,W]
552        Mask array, with values of `1` inside
553        the polygon, and `0` outside.
554
555    """
556    maskImage = np.ones(maskSize, dtype=np.uint8)
557
558    for i in range(len(boundaryPoints)-1):
559        # Have to reverse coordinates
560        cv2.line(maskImage, np.int64(boundaryPoints[i][::-1]), np.int64(boundaryPoints[i+1][::-1]), 255, 1)        
561    cv2.line(maskImage, np.int64(boundaryPoints[-1][::-1]), np.int64(boundaryPoints[0][::-1]), 255, 1)
562                    
563    outsidePoint = (0,0)
564    cv2.floodFill(maskImage, None, outsidePoint, 255)
565
566    maskImage = (maskImage != 255).astype(np.float64)
567
568    if debug:
569        plt.imshow(maskImage)
570        plt.show()
571
572    return maskImage

575def removePadding(image):
576    """
577    Remove padding from an image.
578
579    Parameters
580    ----------
581    image : array-like
582        Image to remove excess padding from.
583
584    Returns
585    -------
586    unpaddedImage : array-like
587        Original image with rows and columns
588        removed if they contained no image data.
589    """
590    # Grayscale if necessary
591    if len(np.shape(image)) == 3:
592        grayImage = np.mean(image, axis=-1)
593    else:
594        grayImage = image
595
596    rowSum = np.sum(grayImage, axis=0)
597    columnSum = np.sum(grayImage, axis=-1)
598
599    # There is probably a much fancier way to do this, but this
600    # is fast enough, so... essentially we just count until the sum
601    # is not zero anymore (ie there is some image data)
602    
603    leftEdge = None
604    for i in range(len(rowSum)):
605        leftEdge = i
606        if rowSum[i] > 0:
607            break
608
609    rightEdge = None
610    for i in range(len(rowSum))[::-1]:
611        rightEdge = i
612        if rowSum[i] > 0:
613            break
614
615    topEdge = None
616    for i in range(len(columnSum)):
617        topEdge = i
618        if columnSum[i] > 0:
619            break
620            
621    bottomEdge = None
622    for i in range(len(columnSum))[::-1]:
623        bottomEdge = i
624        if columnSum[i] > 0:
625            break
626
627    return image[topEdge:bottomEdge,leftEdge:rightEdge]

593        grayImage = np.mean(image, axis=-1)
594    else:
595        grayImage = image
596
597    rowSum = np.sum(grayImage, axis=0)
598    columnSum = np.sum(grayImage, axis=-1)
599
600    # There is probably a much fancier way to do this, but this
601    # is fast enough, so... essentially we just count until the sum
602    # is not zero anymore (ie there is some image data)
603

605    for i in range(len(rowSum)):
606        leftEdge = i
607        if rowSum[i] > 0:
608            break

624        bottomEdge = i
625        if columnSum[i] > 0:
